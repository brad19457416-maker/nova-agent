# Nova Agent v0.3.0 实现方案

**基于决策**:
- 产品定位: A + B (通用框架 + 深度研究专家)
- 工作流: 混合模式 + 细粒度(5-6阶段)
- 技能: 声明式 + 实现分离
- 反模式: 手动编写 + 从反馈学习
- 协作: P0实现Lead/Sub
- 技术栈: YAML + SQLite

---

## 🎯 核心设计原则

1. **不只是框架** - 每个工作流必须可独立运行
2. **避免硬编码** - 配置驱动，声明式定义
3. **开箱即用** - 内置可工作的核心技能

```python
# config.py

import yaml
import json
from pathlib import Path
from typing import Any, Dict, Optional

class ConfigManager:
    """配置管理器 - 支持YAML/JSON"""
    
    def __init__(self, config_dir: str = "./config"):
        self.config_dir = Path(config_dir)
        self.configs: Dict[str, Any] = {}
    
    def load(self, name: str) -> Dict:
        """加载配置"""
        # 尝试YAML
        yaml_path = self.config_dir / f"{name}.yaml"
        if yaml_path.exists():
            with open(yaml_path) as f:
                self.configs[name] = yaml.safe_load(f)
                return self.configs[name]
        
        # 尝试JSON
        json_path = self.config_dir / f"{name}.json"
        if json_path.exists():
            with open(json_path) as f:
                self.configs[name] = json.load(f)
                return self.configs[name]
        
        raise FileNotFoundError(f"Config {name} not found")
    
    def get(self, name: str, key: str, default: Any = None) -> Any:
        """获取配置值"""
        if name not in self.configs:
            self.load(name)
        return self.configs[name].get(key, default)
    
    def set(self, name: str, key: str, value: Any):
        """设置配置值"""
        if name not in self.configs:
            self.configs[name] = {}
        self.configs[name][key] = value
    
    def save(self, name: str):
        """保存配置"""
        path = self.config_dir / f"{name}.yaml"
        with open(path, 'w') as f:
            yaml.dump(self.configs[name], f)
```

### 2. 工作流引擎

```python
# workflow/engine.py

from typing import Dict, List, Any, Callable, Optional
from enum import Enum
import asyncio
from dataclasses import dataclass

class PhaseStatus(Enum):
    PENDING = "pending"
    RUNNING = "running"
    COMPLETED = "completed"
    FAILED = "failed"
    SKIPPED = "skipped"

@dataclass
class PhaseResult:
    """阶段执行结果"""
    phase: str
    status: PhaseStatus
    output: Any = None
    error: str = ""
    metadata: Dict = None

class WorkflowEngine:
    """工作流引擎 - 配置驱动"""
    
    def __init__(self, config_manager: ConfigManager):
        self.config = config_manager
        self.phases: Dict[str, Callable] = {}
        self.workflows: Dict[str, Dict] = {}
        self._register_builtin_phases()
        self._load_builtin_workflows()
    
    def _register_builtin_phases(self):
        """注册内置阶段处理器"""
        self.phases = {
            "clarify": self._phase_clarify,
            "plan": self._phase_plan,
            "search": self._phase_search,
            "expand": self._phase_expand,
            "verify": self._phase_verify,
            "synthesize": self._phase_synthesize,
            "outline": self._phase_outline,
            "draft": self._phase_draft,
            "review": self._phase_review,
            "polish": self._phase_polish,
            "deliver": self._phase_deliver,
        }
    
    def _load_builtin_workflows(self):
        """加载内置工作流"""
        # 从YAML加载
        for workflow_name in ["research", "writing", "code"]:
            try:
                config = self.config.load(f"workflows/{workflow_name}")
                self.workflows[workflow_name] = config
            except FileNotFoundError:
                # 使用默认配置
                self.workflows[workflow_name] = self._get_default_workflow(workflow_name)
    
    def register_phase(self, name: str, handler: Callable):
        """注册自定义阶段"""
        self.phases[name] = handler
    
    def register_workflow(self, name: str, config: Dict):
        """注册自定义工作流"""
        self.workflows[name] = config
    
    async def run(self, workflow_name: str, context: Dict) -> Dict:
        """运行工作流"""
        if workflow_name not in self.workflows:
            raise ValueError(f"Workflow {workflow_name} not found")
        
        workflow_config = self.workflows[workflow_name]
        phases = workflow_config.get("phases", [])
        
        results = []
        ctx = context.copy()
        
        for phase_name in phases:
            # 检查条件
            condition = workflow_config.get("conditions", {}).get(phase_name)
            if condition and not self._evaluate_condition(condition, ctx):
                results.append(PhaseResult(
                    phase=phase_name,
                    status=PhaseStatus.SKIPPED
                ))
                continue
            
            # 执行阶段
            handler = self.phases.get(phase_name)
            if not handler:
                results.append(PhaseResult(
                    phase=phase_name,
                    status=PhaseStatus.FAILED,
                    error=f"Phase handler not found: {phase_name}"
                ))
                continue
            
            try:
                result = await handler(ctx)
                results.append(PhaseResult(
                    phase=phase_name,
                    status=PhaseStatus.COMPLETED,
                    output=result
                ))
                ctx[phase_name] = result
            except Exception as e:
                results.append(PhaseResult(
                    phase=phase_name,
                    status=PhaseStatus.FAILED,
                    error=str(e)
                ))
                # 根据配置决定是否继续
                if not workflow_config.get("continue_on_error", False):
                    break
        
        return {
            "workflow": workflow_name,
            "results": results,
            "final_context": ctx
        }
    
    # 阶段处理器 - 可以被配置覆盖
    async def _phase_clarify(self, ctx: Dict) -> Dict:
        """澄清阶段"""
        task = ctx.get("task", "")
        # 使用LLM澄清任务
        return {"clarified_task": f"澄清后的任务: {task}"}
    
    async def _phase_plan(self, ctx: Dict) -> Dict:
        """规划阶段"""
        return {"plan": "任务计划"}
    
    # ... 其他阶段处理器
```

```python
# skills/base.py

import yaml
from typing import Dict, List, Any, Callable, Optional
from pathlib import Path
from dataclasses import dataclass, field

@dataclass
class SkillConfig:
    """技能配置"""
    name: str
    description: str
    version: str = "1.0.0"
    phases: List[Dict] = field(default_factory=list)
    antipatterns: List[str] = field(default_factory=list)
    references: List[str] = field(default_factory=list)
    tools: List[str] = field(default_factory=list)
    metadata: Dict = field(default_factory=dict)

class BaseSkill:
    """技能基类"""
    
    def __init__(self, config: SkillConfig):
        self.config = config
        self.phase_handlers: Dict[str, Callable] = {}
    
    async def execute_phase(self, phase_name: str, context: Dict) -> Dict:
        """执行单个阶段"""
        if phase_name not in self.phase_handlers:
            raise ValueError(f"Phase {phase_name} not found in skill {self.config.name}")
        
        handler = self.phase_handlers[phase_name]
        return await handler(context)
    
    async def execute(self, context: Dict) -> Dict:
        """执行完整技能"""
        results = {}
        
        for phase in self.config.phases:
            phase_name = phase.get("name")
            result = await self.execute_phase(phase_name, context)
            results[phase_name] = result
            context[f"{self.config.name}_{phase_name}"] = result
        
        return results
    
    def get_antipatterns(self) -> List[str]:
        """获取反模式"""
        return self.config.antipatterns
    
    def get_references(self) -> List[str]:
        """获取参考文档"""
        return self.config.references


class SkillLoader:
    """技能加载器 - YAML + 实现"""
    
    def __init__(self, skills_dir: str = "./skills"):
        self.skills_dir = Path(skills_dir)
        self.skills: Dict[str, BaseSkill] = {}
    
    def load(self, skill_name: str) -> BaseSkill:
        """加载技能"""
        # 加载YAML配置
        config_path = self.skills_dir / f"{skill_name}.yaml"
        with open(config_path) as f:
            config_data = yaml.safe_load(f)
        
        # 创建配置对象
        config = SkillConfig(**config_data)
        
        # 加载实现
        impl_path = self.skills_dir / f"{skill_name}_impl.py"
        if impl_path.exists():
            # 动态导入实现
            import importlib.util
            spec = importlib.util.spec_from_file_location(
                f"{skill_name}_impl", 
                impl_path
            )
            module = importlib.util.module_from_spec(spec)
            spec.loader.exec_module(module)
            
            # 获取Skill类
            skill_class = getattr(module, f"{skill_name.replace('-', '_').title()}Skill")
            skill = skill_class(config)
        else:
            skill = BaseSkill(config)
        
        self.skills[skill_name] = skill
        return skill
    
    def list_skills(self) -> List[str]:
        """列出所有技能"""
        return [p.stem for p in self.skills_dir.glob("*.yaml")]
```

```python
# collaboration/lead_sub.py

import asyncio
from typing import Dict, List, Any, Optional
from enum import Enum
from dataclasses import dataclass

class AgentRole(Enum):
    LEADER = "leader"
    SUB = "sub"
    REVIEWER = "reviewer"
    RESEARCHER = "researcher"

class AgentState(Enum):
    IDLE = "idle"
    WAITING = "waiting"
    WORKING = "working"
    COMPLETED = "completed"
    FAILED = "failed"

@dataclass
class Agent:
    """Agent定义"""
    id: str
    role: AgentRole
    state: AgentState
    context: Dict = None

@dataclass
class Task:
    """任务定义"""
    id: str
    description: str
    status: str = "pending"
    assigned_to: Optional[str] = None
    result: Any = None

class LeadSubCollaboration:
    """Lead/Sub协作实现"""
    
    def __init__(self, llm_client, max_subs: int = 5):
        self.llm = llm_client
        self.max_subs = max_subs
        self.agents: Dict[str, Agent] = {}
        self.tasks: Dict[str, Task] = {}
    
    async def execute(self, task: str, config: Dict) -> Dict:
        """执行Lead/Sub协作"""
        # 1. Leader分解任务
        subtasks = await self._decompose_task(task, config)
        
        # 2. 创建Sub Agents
        subs = await self._create_sub_agents(subtasks)
        
        # 3. 并行执行
        results = await asyncio.gather(*[
            self._execute_sub(sub, task) for sub in subs
        ])
        
        # 4. Leader汇总
        final_result = await self._synthesize_results(task, results)
        
        return {
            "task": task,
            "subtasks": len(subtasks),
            "results": results,
            "final_result": final_result
        }
    
    async def _decompose_task(self, task: str, config: Dict) -> List[str]:
        """Leader分解任务"""
        prompt = f"""将以下任务分解为可独立执行的子任务。

任务: {task}

要求:
1. 每个子任务应该可以独立完成
2. 子任务之间有明确的边界
3. 总数不超过{self.max_subs}个

子任务列表:"""
        
        response = await self.llm.complete(prompt)
        # 解析子任务
        subtasks = [line.strip() for line in response.split('\n') if line.strip()]
        return subtasks[:self.max_subs]
    
    async def _create_sub_agents(self, subtasks: List[str]) -> List[Agent]:
        """创建Sub Agents"""
        agents = []
        for i, task in enumerate(subtasks):
            agent = Agent(
                id=f"sub_{i}",
                role=AgentRole.SUB,
                state=AgentState.IDLE,
                context={"task": task}
            )
            agents.append(agent)
            self.agents[agent.id] = agent
        return agents
    
    async def _execute_sub(self, agent: Agent, parent_context: Dict) -> Dict:
        """执行子任务"""
        agent.state = AgentState.WORKING
        
        # 执行任务
        task = agent.context.get("task", "")
        result = await self.llm.complete(task)
        
        agent.state = AgentState.COMPLETED
        agent.context["result"] = result
        
        return {
            "agent_id": agent.id,
            "task": task,
            "result": result
        }
    
    async def _synthesize_results(self, original_task: str, results: List[Dict]) -> str:
        """Leader汇总结果"""
        prompt = f"""原始任务: {original_task}

子任务结果:
{chr(10).join([f"- {r['result']}" for r in results])}

请汇总所有子任务结果，给出最终答案:"""
        
        return await self.llm.complete(prompt)
```

```python
# storage/sqlite_store.py

import sqlite3
import json
from pathlib import Path
from typing import Dict, List, Any, Optional
from datetime import datetime

class SQLiteStore:
    """SQLite存储"""
    
    def __init__(self, db_path: str = "./data/nova.db"):
        self.db_path = db_path
        Path(db_path).parent.mkdir(parents=True, exist_ok=True)
        self.conn = sqlite3.connect(db_path)
        self._init_tables()
    
    def _init_tables(self):
        """初始化表"""
        self.conn.executescript('''
            CREATE TABLE IF NOT EXISTS workflows (
                id INTEGER PRIMARY KEY,
                name TEXT UNIQUE NOT NULL,
                config TEXT NOT NULL,
                created_at TEXT NOT NULL,
                updated_at TEXT NOT NULL
            );
            
            CREATE TABLE IF NOT EXISTS skills (
                id INTEGER PRIMARY KEY,
                name TEXT UNIQUE NOT NULL,
                config TEXT NOT NULL,
                version TEXT NOT NULL,
                created_at TEXT NOT NULL
            );
            
            CREATE TABLE IF NOT EXISTS antipatterns (
                id INTEGER PRIMARY KEY,
                name TEXT NOT NULL,
                description TEXT,
                category TEXT NOT NULL,
                severity TEXT,
                created_at TEXT NOT NULL
            );
            
            CREATE TABLE IF NOT EXISTS executions (
                id INTEGER PRIMARY KEY,
                workflow_name TEXT NOT NULL,
                input TEXT NOT NULL,
                output TEXT,
                status TEXT NOT NULL,
                created_at TEXT NOT NULL,
                duration_ms INTEGER
            );
            
            CREATE TABLE IF NOT EXISTS feedback (
                id INTEGER PRIMARY KEY,
                execution_id INTEGER,
                rating INTEGER,
                comment TEXT,
                antipatterns_found TEXT,
                created_at TEXT NOT NULL,
                FOREIGN KEY(execution_id) REFERENCES executions(id)
            );
        ''')
        self.conn.commit()
    
    def save_workflow(self, name: str, config: Dict):
        """保存工作流"""
        now = datetime.now().isoformat()
        self.conn.execute('''
            INSERT OR REPLACE INTO workflows (name, config, created_at, updated_at)
            VALUES (?, ?, ?, ?)
        ''', (name, json.dumps(config), now, now))
        self.conn.commit()
    
    def get_workflow(self, name: str) -> Optional[Dict]:
        """获取工作流"""
        cursor = self.conn.execute(
            'SELECT config FROM workflows WHERE name = ?', (name,)
        )
        row = cursor.fetchone()
        return json.loads(row[0]) if row else None
    
    def save_execution(self, workflow_name: str, input_data: Dict, 
                      output: Any, status: str, duration_ms: int):
        """保存执行记录"""
        now = datetime.now().isoformat()
        self.conn.execute('''
            INSERT INTO executions (workflow_name, input, output, status, created_at, duration_ms)
            VALUES (?, ?, ?, ?, ?, ?)
        ''', (workflow_name, json.dumps(input_data), json.dumps(output), status, now, duration_ms))
        self.conn.commit()
    
    def add_feedback(self, execution_id: int, rating: int, comment: str):
        """添加反馈"""
        now = datetime.now().isoformat()
        self.conn.execute('''
            INSERT INTO feedback (execution_id, rating, comment, created_at)
            VALUES (?, ?, ?, ?)
        ''', (execution_id, rating, comment, now))
        self.conn.commit()
    
    def get_stats(self) -> Dict:
        """获取统计信息"""
        cursor = self.conn.execute('''
            SELECT 
                COUNT(*) as total_executions,
                SUM(CASE WHEN status = 'completed' THEN 1 ELSE 0 END) as completed,
                AVG(duration_ms) as avg_duration
            FROM executions
        ''')
        row = cursor.fetchone()
        
        return {
            "total_executions": row[0] or 0,
            "completed": row[1] or 0,
            "avg_duration_ms": row[2] or 0
        }
```
